64  {
65  Type place = (Type)1 << ( sizeof (Type) * 8 - 2 ); // calculated by precompiler = same runtime as: place = 0x40000000
66  while( place > iN )
67  place /= 4; // optimized by complier as place >>= 2
68

70

72  {
73  if( iN >= root+place )
74  {
75  iN -= root+place;
76  root += place * 2;
77  }
78 
79  root /= 2;
80  place /= 4;
81  }
82
